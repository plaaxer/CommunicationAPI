136 static int receive(Buffer * buf, Address* from, void* data, unsigned int size)
137 {
138 if (!buf || !from || !data) {
139 return -1;
140 }
141
142 // payload structures
143 Ethernet::Frame* frame = buf->data();
144 Packet* packet = reinterpret_cast<Packet*>(frame->data);
145 PortHeader* proto_header = packet->portheader();
146
147 *from = Address(frame->header.shost, proto_header->sport());
148
149 // Frame data includes PortHeader
150 unsigned int data_size_in_packet = frame->data_length - sizeof(PortHeader);
151 unsigned int bytes_to_copy = std::min(size, data_size_in_packet);
152
153 /*
154 We just return the actual payload without the PortHeader as the addressing is saved
155 on the Address* from parameter.
156 */
157
158 std::memcpy(data, packet->template data<void>(), bytes_to_copy);
159
160 return bytes_to_copy;
161 }
