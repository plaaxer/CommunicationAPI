223
224 struct sembuf acquire_full = {FULL_SEM, -1, SEM_UNDO};
225
226 if (semop(_sem_id, &acquire_full, 1) == -1) {
227 perror("semop (WAIT ON FULL SEM) failed on receive");
228 return -1;
229 }

232
233 SharedBlock* block = static_cast<SharedBlock*>(_shared_block);
234
235 Ethernet::Frame* frame = &block->frame_buffer;
236
237 int data_size = frame->data_length;
238 int total_size = data_size + sizeof(frame->header);
239 int min_copied = std::min(total_size, max_size);
240
241 memcpy(frame_buffer, frame, min_copied);
242
243 // END OF PROTECTED ZONE
244
245 struct sembuf release_empty = {EMPTY_SEM, +1, SEM_UNDO};
246
247 if (semop(_sem_id, &release_empty, 1) == -1) {
248 perror("semop (RELEASE ON EMPTY SEM) failed on receive");
249 return -1;
250 }

252 return min_copied;

269 void _cleanupAndThrow(const std::string& errorMessage) {
270 std::cerr << "CRITICAL ERROR during ShmEngine construction: " << errorMessage << " - " << strerror(errno) << std::endl;
271 std::cerr << "Performing cleanup..." << std::endl;
272
273 // cleanup in reverse order of acquisition
274
275 // Detach shared memory if it was attached
276 if (_shared_block != nullptr && _shared_block != (void*)-1) {
277 shmdt(_shared_block);
278 }
279
280 // Remove shared memory segment if it was created
281 if (_shared_id != -1) {
282 shmctl(_shared_id, IPC_RMID, nullptr);
283 }
284
285 // Remove semaphore set if we were the one who created it (no race conditions?).
286 if (_sem_id != -1) {
287 semctl(_sem_id, 0, IPC_RMID, nullptr);
288 }
289
290 throw std::runtime_error(errorMessage);
291 }
